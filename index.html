' expanded' : ''; // Only first phase expanded by default const expandedContent = phaseCount === 1 ? ' expanded' : ''; currentPhaseSection = `
▼

${escapeHtml(phaseMatch[1])}

`; continue; } // Handle content within a phase if (currentPhaseSection && !phaseMatch) { if (taskMatch) { const isChecked = taskMatch[1].toLowerCase() === 'x'; const taskText = taskMatch[2].trim(); const taskId = generateTaskId(taskText); if (!inTaskList) { inTaskList = true; taskListHtml = '
    '; } taskListHtml += `
    • `; } else { // End of task list if (inTaskList) { taskListHtml += '
'; currentPhaseSection += taskListHtml; inTaskList = false; taskListHtml = ''; } // Horizontal rule if (line.trim() === '---') { currentPhaseSection += '
'; } // Sub-headings within phase else if (line.startsWith('### ')) { currentPhaseSection += `

${escapeHtml(line.replace(/^### /, ''))}

`; } // Blockquotes else if (line.startsWith('> ')) { currentPhaseSection += `
${escapeHtml(line.replace(/^> /, ''))}
`; } // Regular paragraphs else if (line.trim() && !line.startsWith('#')) { currentPhaseSection += `

${escapeHtml(line)}

`; } } } else if (!currentPhaseSection && !phaseMatch) { // Content before any phase (overview, intro) if (taskMatch) { const isChecked = taskMatch[1].toLowerCase() === 'x'; const taskText = taskMatch[2].trim(); const taskId = generateTaskId(taskText); if (!inTaskList) { inTaskList = true; taskListHtml = '

    • `; } else { if (inTaskList) { taskListHtml += '
'; html += taskListHtml; inTaskList = false; taskListHtml = ''; } if (line.trim() === '---') { html += '
